def own_split(X, y):
  scaler = StandardScaler()
  X_train, X_test, y_train, y_test = train_test_split(X, y, train_size=0.8, random_state=42)
  return (
    pd.DataFrame(scaler.fit_transform(X_train), columns=X.columns), 
    pd.DataFrame(scaler.transform(X_test), columns=X.columns), 
    y_train, 
    y_test)

def pvi():
  res = {
    model_name : {}
    for model_name in models.keys()
  }
  no_noise = {}
  X_test_ = X_test.copy()
  for model_name, model in models.items():
    pred = model.predict(X_test_)
    no_noise[model_name] = balanced_accuracy_score(pred, y_test)

  for feature in X_test.columns:
    # apply the permutation
    X_test_[feature] = np.random.permutation(X_test_[feature])

    for model_name, model in models.items():
      pred = model.predict(X_test_)
      res[model_name][feature] = balanced_accuracy_score(pred, y_test) / no_noise[model_name]

    # undo the permutation
    X_test_[feature] = X_test[feature].copy()
  return pd.DataFrame(res), no_noise
  
var_imp, no_noise = pvi()
